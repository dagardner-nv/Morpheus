list(APPEND CMAKE_MESSAGE_CONTEXT "tests")

find_package(pybind11 REQUIRED)

include(GoogleTest)

add_library(
    morpheus_test_utilities
  test_utils/common.cpp
)

target_link_libraries(
  morpheus_test_utilities
  PRIVATE
    GTest::gtest
    morpheus
)

function(add_morpheus_test TEST_NAME TEST_FILES)
  add_executable(
    test_${TEST_NAME}
    ${TEST_FILES}
  )

  target_link_libraries(
    test_${TEST_NAME}
    PRIVATE
      GTest::gtest
      GTest::gtest_main
      matx::matx
      morpheus
      morpheus_test_utilities
      pybind11::embed
  )

  gtest_discover_tests(test_${TEST_NAME})

  set_target_properties(test_${TEST_NAME}
    PROPERTIES
      INSTALL_RPATH "$ORIGIN/.."
      CUDA_STANDARD 17
      CUDA_STANDARD_REQUIRED ON
  )

  install(
    TARGETS
      test_${TEST_NAME}
    RUNTIME DESTINATION
      "${MORPHEUS_LIB_INSTALL_DIR}/tests"
    COMPONENT Wheel
  )
endfunction()
